# HTML Basic Template

```html
<!doctype html>
<html>
  <head>
    <meta charset="utf-8">
    <meta name="description" content="A simple empty HTML page">
    <meta name="author" content="Author Name">
    <meta name="viewport" content="width=device-width, initial-scale=1">
    <title>HTML Basic Template</title>

    <link rel="icon" href="assets/img/favicon.png">
  </head>

    <main>
      <header>
        <h1>Heading 1</h1>
        <h2>Heading 2</h2>
      </header>
      <nav>
        <ul>
          <li>Nav Item</li>
          <li>Nav Item</li>
          <li>Nav Item</li>
        </ul>
      </nav>
      <section>
        <h3>Heading 3</h3>
        <p>Section paragraph.</p>
        <article>
          <h4>Article Heading</h4>
          <p>Article paragraph.</p>
        </article>
      </section>
      <aside>
        <h4>Aside Heading</h4>
        <p>Aside paragraph.</p>
      </aside>
      <footer>
        <h4>Footer Heading</h4>
        <p>Footer paragraph</p>
      </footer>
    </main>
    <script src="assets/js/script.js"></script>
  </body>
</html>
```
